export function formatNum(strNum) {
  if (strNum.length <= 3) {
    return strNum;
  }
  if (!/^(\+|-)?(\d+)(\.\d+)?$/.test(strNum)) {
    return strNum;
  }
  const a = RegExp.$1;
  let b = RegExp.$2;
  const c = RegExp.$3;
  const re = new RegExp();
  re.compile("(\\d)(\\d{3})(,|$)");
  while (re.test(b)) {
    b = b.replace(re, "$1,$2$3");
  }
  return `${a}${b}${c}`;
}

//计算环形图百分比
function piePercenter(arr, name) {
  let total = 0;
  let data = {
    percent: 0,
    currentValue: 0,
  };
  if (arr && arr.length) {
    for (let key = 0; key < arr.length; key++) {
      total += arr[key].value;
      if (name === arr[key].name) {
        data.currentValue = arr[key].value;
      }
    }
    data.percent = ((data.currentValue / total) * 100).toFixed(2);
    return data;
  }
}

function barPercenter(arr, value) {
  var total = arr.reduce((pre, cur) => {
    return pre + cur;
  }, 0);
  if (total === 0) {
    return "0.00" + "%";
  } else {
    return ((value / total) * 100).toFixed(2) + "%";
  }
}
