# 视频分片上传

JavaScript 利用 Blob 进行分片上传 (opens new window)

$("#upload-vedio").on("change", function (e) {
    var file = e.target.files[0];
    
    var maxSize=100 *1024 *1824;
    if(file.size >maxSize){
        GlobalTip.TipMessageBox("请上传100M以内的视频");
        return;
    }
    var typeArr=['mp4','rmvb','avi'];
    var type=file.type.split('/')[1];
    if(!typeArr.includes(type)){
        GlobalTip.TipMessageBox("视频格式不正确！");
        return;
    }
	uploadFile("/upload", file, function (err, res) {
		if (err) {
			return GlobalTip.TipMessageBox("上传失败，请重新上传!");
		}
		GlobalTip.TipMessageBox("上传成功");
	});
});

function uploadFile(url, blob, callback) {
	var csrf = $("meta[name='csrf-token']").attr("content");
	var perFileSize = 2 * 1024 * 1024; //每一个分快的大小
	var blobParts = Math.ceil(blob.size / perFileSize);

	for (var i = 0; i < blobParts; i++) {
		(function (i) {
			var formData = new FormData();
			var _blob = blob.slice(i * perFileSize, (i + 1) * perFileSize);
			formData.append("_blob", _blob);
			formData.append("filename", blob.name);
			formData.append("index", i + 1);
			formData.append("total", blobParts);
			formData.append("csrf", csrf);
			$.ajax({
				type: "POST",
				url: url,
				data: formData,
				processData: false,
				contentType: false,
				success: function (data) {
					$(".upload-vedio").hide();
					$(".show-video").html(`
                   <video
                        id="my-player"
                        class="video-js vjs-big-play-centered"
                        controls
                        preload="auto"
                        poster="../image/cover-image.png"
                        width="240" height="180"
                        data-setup='{}'>
                        <source src="video.mp4" type="video/mp4"></source>
                        
                        <p class="vjs-no-js">
                        请开启Javascript脚本以便
                        <a href="http://videojs.com/html5-video-support/" target="_blank">
                        支持HTML5 video播放器
                        </a>
                        </p>
                   </video>
                    `);
				},
				fail: function (err) {
					callback(err, null);
				},
			});
		})(i);
	}
}

# 上传附件

$("#uploadFile").on("change", function () {
        var _csrf = $('meta[name="csrf-token"]').attr("content");
        var files = $("#uploadFile")[0].files,
            typeArr = [".pdf", ".doc", ".xls", ".ppt", ".txt", ".docx", ".pptx", ".xlsx"];
        var formData = new FormData();
        for (var i = 0; i < files.length; i++) {
            var point = files[i]["name"].lastIndexOf(".");
            var type = files[i]["name"].substr(point).toLowerCase();
            if (typeArr.indexOf(type) == -1) {
                GlobalTip.WarningTipBox("文件格式不匹配，请重新上传", "确定");
                return false;
            }
            var img_size = files[i]['size'];
            var max_size = 10 * 1024 * 1024; //文件大小
            if (img_size > max_size) {
                GlobalTip.WarningTipBox("文件太大，请重新上传", "确定");
                return false;
            }
            if (files.length > 9) {
                GlobalTip.WarningTipBox("文件超出限制,请重新上传", "确定");
                return false;
            }
            var name = files[i].name;
            var file = files[i];

            formData.append("fileFile" + i, file);
            formData.append("name" + i, name);
        }
        formData.append("count", files.length);
        formData.append("_csrf", _csrf);
        $.ajax({
            url: $("#attr-url").val(),
            type: "POST",
            data: formData,
            processData: false,
            contentType: false,
            success: function (data) {
                if (data.code == 1) {
                    var docs1=data.docs;
                    for (var p in docs1) {//遍历json数组时，这么写p为索引，0,1
                        $(".fileShow ul").append(
                            "<li>" +
                            '<input type="hidden" name="attr_ids[]" value="' + docs1[p].id +
                            '">' +
                            "<span>" +
                            '<i class="file-list-icon"></i>' +
                            '<span class="file-name">' +docs1[p].name+
                            "</span> " +
                            "</span>" +
                            '<span class="file-action-icons">' +
                            '<i class="file-delete" data-id="'+ docs1[p].id +'"></i>' +
                            "</span>" +
                            "</li>")
                    }
                }
            }
        });
    });

function base64data(fileID,clackFn){
   var size=arguments[2]?arguments[2]:3;
   var file=document.getElementById(fileID);
   var arr = ["gif","jpeg","jpg","png"];
   var fileSize=1024*1024*parseInt(size);
   var fileContentType = file.value.match(/^(.*)(\.)(.{1,8})$/)[3]; //文件类型
   if(file.files && file.files[0]){
       var f_size=file.files[0].size;
       if(f_size >fileSize){
        GlobalTip.askConfirmTipBox(false,"图片太大，请重新上传",'确定');
        file.value="";
        return false;
       }else{
           if(arr.join(",").toLowerCase().indexOf(fileContentType.toLowerCase()) !=-1){
               var reader=new FileReader();
               reader.onload=function(evt){
                   clackFn(evt.target.result);
               }
               reader.readAsDataURL(file.files[0]);
           }else{
            GlobalTip.askConfirmTipBox(false,"文件格式不匹配，请重新上传",'确定'); 
           }
       }
   }else{
    GlobalTip.askConfirmTipBox(false,'请切到高级浏览器,再进行图片上传','确定');
   }
}

# 全选/反选

$("#checkAll").on("click", function () {
		var checked = $(this).prop("checked");
		$("tbody tr td input[type='checkbox']").prop("checked", checked);
	});

	//选择
	$(".table-wrap table tr td :checkbox").on("click", function () {
		$(".table-wrap table tr td input:checkbox").length === $(".table-wrap table tr td input:checked").length ? $("#checkAll").prop("checked", true) : $("#checkAll").prop("checked", false);
	});
