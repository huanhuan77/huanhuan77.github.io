import {token, getSessionKey} from "../../api/auth.js"

Component({
  /**
   * 组件的属性列表
   */
  properties: {},

  /**
   * 组件的初始数据
   */
  data: {
    settingShow: false
  },
  attached: function () { //检测有没有缓存手机号码
    if (wx.getStorageSync('token')) {
      //已经登录
    } else {
      this.check()
    }
  },
  /**
   * 组件的方法列表
   */

  //显示登录或授权提示
  methods: {
    /**
     * 登录相关
     */
    showLoginModal: function () {
      this.setData({
        settingShow: true
      })
    },
    //判断是否登录
    check: function (cb) {
      var that = this;
      // that.login();
      that.showLoginModal()
    },
    login: function () {
      var that = this;
      //调用登录接口
      wx.login({
        success(res) {
          if (res.code) {
            // 发送 res.code 到后台换取 openId, sessionKey, unionId
            getSessionKey({code: res.code})
              .then(_res => {
                //成功返回数据后，将sessionKey值存储到localStorage中
                wx.setStorageSync('session_key', _res.data.data.session_key)
                that.getToken()
              })
              .catch(err => {
                wx.setStorageSync('session_key', '');
                that.showLoginModal();
              })
          } else {
            that.showLoginModal()
          }
        }
      })
    },

    //授权
    getPhoneNumber(e) {
      const that = this
      that.setData({
        settingShow: false
      })
      if (e.detail.errMsg === 'getPhoneNumber:ok') {
        wx.setStorageSync("phone_key", e.detail);

        this.login()
        // this.getToken()
      }else{
        wx.showModal({
          title:'提示',
          content: '必须登录才能操作哦!',
          confirmText:'返回登录',
          confirmColor:' #6ed4c1',
          showCancel:false,
          success (res) {
            if (res.confirm) {
              that.showLoginModal()
            }
          }
        })
      }
    },
    //获取token
    getToken() {
      let data = {
        encryptedData: wx.getStorageSync('phone_key').encryptedData,
        iv: wx.getStorageSync('phone_key').iv,
        session_key: wx.getStorageSync('session_key'),
      }
      token({
        encryptedData: data.encryptedData,
        sessionKey: data.session_key,
        iv: data.iv
      })
        .then((res) => {
          wx.setStorageSync('token', res.data.access_token)
          //刷新当前页面
          var pages = getCurrentPages();
          var prevPage = pages[pages.length - 1]; //当前页面)
          prevPage.onPullDownRefresh(); //重新刷新页面
        })
    }
  }
})

# 封装wx.request (opens new window)

  • /utils/request.js
import Toast from '../miniprogram_npm/vant-weapp/toast/toast';
const request = options => {
  wx.showLoading({
    title: '拼命加载中...',
  })

  return new Promise((resolve, reject) => {
    const {
      data,
      method,
      header
    } = options
    if (data && method != 'get') {
      options.data = JSON.stringify(data)
    }

    //发起请求
    wx.request({
      ...options,
      success: function(res) {
        if (res.data.code == 0) {
          resolve(res.data)
        } else if (res.data.code == 1) {
          Toast.fail('服务器错误!');
          reject(res.data)
        } else if(res.data.code==401){
          reject(res.data)
          Toast.fail(res.data.msg);
        }
      },
      //请求失败
      fail: function(res) {
        Toast.fail('网络错误!');
        reject(res.data)
      },
      //请求完成  无论失败 成功都会执行
      complete: function() {
        wx.hideLoading()
      }
    })
  })
}

export default request

import Toast from '../miniprogram_npm/vant-weapp/toast/toast';
import url from '../config.js'

var Fly = require("../lib/flyio/wx.js") //wx.js为您下载的源码文件
var fly = new Fly; //创建fly实例

//获取白名单
import whiteList from "./whiteList";

fly.config.baseURL = url.API_URL //设置baseUrl
fly.config.timeout = 60000 //设置过期 时间

//请求拦截器
fly.interceptors.request.use((config) => {
  wx.showLoading({
    title: '拼命加载中...',
    mask: true
  })

  let Authorization = wx.getStorageSync('token')
  //登录接口  不添加Authorization
  if (whiteList.nullHeaderToken.indexOf(config.url) !== -1) {
    Authorization = ''
  } else if (whiteList.getToken.indexOf(config.url) !== -1) {  //获取token  添加授权代码
    Authorization = 'Basic Ymk6Ymk='
  } else {
    Authorization = `Bearer ${wx.getStorageSync('token')}`
  }

  if (Authorization) {
    config.headers.Authorization = Authorization
  }

  return config;
})

fly.interceptors.response.use((response) => {
    wx.hideLoading()
    const {
      code
    } = response.data

    switch (code) {
      case 0:
        return response.data;

      case 1:
        Toast.fail(res.msg);
        return response.data;

      default:
        Toast.fail('服务器错误!');
        break;
    }
  },
  (err, promise) => {
    //发生网络错误后会走到这里
    wx.hideLoading()

    switch (err.status) {
      case 404:
        Toast.fail('请求地址出错!');
        wx.clearStorageSync();
        break;

      case 500:
        Toast.fail('服务器出错!')
        break;

      default:

    }
    return promise.reject(err.response)
  })

export default fly

import url from '../config.js'
import fly from '../request/api_fly.js'
export function getSessionKey(params) {
  return fly.request({
    baseURL: url.AUTH_URL,
    url: 'wxApp/auth/code2Session',
    method: 'get',
    params,
  })
}

export function token(data) {
  return fly.request({
    baseURL: url.AUTH_URL,
    url: 'auth/app/token',
    body:data,
    method: 'post'
  })
}

# 下拉刷新 上拉加载

 /**
   * 页面相关事件处理函数--监听用户下拉动作
   */
  onPullDownRefresh: function() {
    wx.showNavigationBarLoading(); //在标题栏中显示加载

    //回到顶部
    wx.pageScrollTo({
      scrollTop: 0,
      selector: ".main"
    });

    this.clearCache();
    this.getOrderList(); //获取数据
  },
      
        /**
   * 页面上拉触底事件的处理函数
   */
  onReachBottom: function() {
    if (this.data.list.length < this.data.total) {
      this.data.pageNumber++;
      this.setData({
        pageNumber: this.data.pageNumber
      });

      this.getOrderList();
    }
  },

      
    //重置数据
  clearCache() {
    this.setData({
      list: [],
      pageNumber: 1
    });
  },
      
    //获取订单列表
  getOrderList() {
    myOrder({
      pageSize: this.data.pageSize,
      pageNumber: this.data.pageNumber,
      storeNo: "440114018",
      state: this.data.tabActive
    }).then(res => {
      wx.hideNavigationBarLoading(); //完成停止加载
      wx.stopPullDownRefresh(); //停止下拉刷新

      var list = res.data.myOrderList;
      this.setData({
        list: this.data.list.concat(list),
        total: res.data.totalCount
      });

      //没有数据
      if (res.data.totalCount == 0) {
        this.setData({
          loading: false,
          finished: true
        });
        //数据加载完毕
      } else if (this.data.list.length == this.data.total) {
        this.setData({
          loading: false,
          finished: true
        });
      }else{
        this.setData({
          loading: true,
          finished: false
        });
      }

    });
  }
