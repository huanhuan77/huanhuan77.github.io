# 上传附件

  • 异步上传
$("#uploadFile").on("change", function () {
  var _csrf = $('meta[name="csrf-token"]').attr("content");
  var files = $("#uploadFile")[0].files,
    typeArr = [
      ".pdf",
      ".doc",
      ".xls",
      ".ppt",
      ".txt",
      ".docx",
      ".pptx",
      ".xlsx",
    ];

  var formData = new FormData();
  for (var i = 0; i < files.length; i++) {
    var point = files[i]["name"].lastIndexOf(".");
    var type = files[i]["name"].substr(point).toLowerCase();
    if (typeArr.indexOf(type) == -1) {
      GlobalTip.WarningTipBox("文件格式不匹配，请重新上传", "确定");
      return false;
    }
    var img_size = files[i]["size"];
    var max_size = 10 * 1024 * 1024; //文件大小
    if (img_size > max_size) {
      GlobalTip.WarningTipBox("文件太大，请重新上传", "确定");
      return false;
    }
    if (files.length > 9) {
      GlobalTip.WarningTipBox("文件超出限制,请重新上传", "确定");
      return false;
    }
    var name = files[i].name;
    var file = files[i];

    formData.append("fileFile" + i, file);
    formData.append("name" + i, name);
  }
  formData.append("count", files.length);
  formData.append("_csrf", _csrf);
  $.ajax({
    url: $("#attr-url").val(),
    type: "POST",
    data: formData,
    processData: false,
    contentType: false,
    success: function (data) {
      if (data.code == 1) {
        var docs1 = data.docs;
        for (var p in docs1) {
          //遍历json数组时，这么写p为索引，0,1
          $(".fileShow ul").append(
            "<li>" +
              '<input type="hidden" name="attr_ids[]" value="' +
              docs1[p].id +
              '">' +
              "<span>" +
              '<i class="file-list-icon"></i>' +
              '<span class="file-name">' +
              docs1[p].name +
              "</span> " +
              "</span>" +
              '<span class="file-action-icons">' +
              '<i class="file-delete" data-id="' +
              docs1[p].id +
              '"></i>' +
              "</span>" +
              "</li>"
          );
        }
        // 统一左右高度
        getHeight();
      }
    },
  });
});

  • file 文件流上传
/**
 * [enterUpload 上传文件]
 * @param  {[type]} fileID  [选取文件的ID]
 * @param  {[type]} typeArr [限制上传文件的类型]
 * @param  {[type]} nameId  [显示文件名字的标签id]
 * @return {[type]}         [size：限制的大小（单位M）]
 */
function enterUpload(fileID, typeArr, nameId) {
  var size = arguments[3] ? arguments[3] : 2;
  var file = document.getElementById(fileID);
  size = 1024 * 1024 * parseInt(size);
  var fileContentType = file.value.match(/^(.*)(\.)(.{1,8})$/)[3]; //文件类型
  var fileName = file.value.split("\\").pop(); // 文件名字
  var imgType = ["gif", "jpeg", "jpg", "png", "bmp"];
  $("#" + nameId)
    .siblings(".fileWrap")
    .find("span.form_tips")
    .removeClass("error_tips")
    .removeClass("right_tips")
    .html("");
  if (file.files && file.files[0]) {
    var f_size = file.files[0].size; //文件大小
    if (
      typeArr.join(",").toLowerCase().indexOf(fileContentType.toLowerCase()) !=
      -1
    ) {
      //检索上传的文件类型是不是在允许的arr数组之内
      if (f_size > size) {
        GlobalTip.WarningTipBox("文件太大，请重新上传", "确定");
        file.value = "";
        return false;
      } else {
        if (
          imgType
            .join(",")
            .toLowerCase()
            .indexOf(fileContentType.toLowerCase()) != -1
        ) {
          // 图片base64
          var reader = new FileReader();
          reader.onload = function (evt) {
            // $("#"+fileID).siblings('input[type="hidden"]').val(evt.target.result);
            $("#" + nameId)
              .siblings(".fileWrap")
              .find('input[type="hidden"]')
              .val(evt.target.result);
          };
          reader.readAsDataURL(file.files[0]);
        } else {
          $("#" + nameId)
            .siblings(".fileWrap")
            .find('input[type="hidden"]')
            .val("");
        }
        $("#" + nameId)
          .css("display", "block")
          .html(fileName);
        $("#" + nameId)
          .siblings(".fileExplain")
          .css("display", "none");
      }
    } else {
      GlobalTip.WarningTipBox("文件格式不匹配，请重新上传", "确定");
    }
  } else {
    //兼容IE
    GlobalTip.WarningTipBox("请切到高级浏览器,再进行图片上传", "确定");
  }
}

//调用
enterUpload(
  "imgUp01",
  ["gif", "jpeg", "jpg", "png", "bmp"],
  "uploadName01",
  50
);
